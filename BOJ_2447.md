``` python
def algorithm(n) :
    global arr

    if n == 3 :
        arr[0][:3] = arr[2][:3] = [1]*3
        arr[1][:3] = [1, 0, 1]
        return

    sub = n//3 
    algorithm(sub) # Recursion
    for i in range(3) :
        for j in range(3) :
            if i == 1 and j == 1 : 
                continue
            for k in range(sub) :
                arr[sub*i+k][sub*j:sub*(j+1)] = arr[k][:sub]

n = int(input())

arr = [[0 for i in range(n)] for i in range(n)]

algorithm(n)

for i in arr :
    for j in i :
        if j :
            print('*', end = '')
        else :
            print(' ', end = '')
    print()
```
